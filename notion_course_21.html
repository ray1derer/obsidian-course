class FolderWatcher {
    plugin: Plugin;
    watchedFolder: string;
    
    constructor(plugin: Plugin, folder: string) {
        this.plugin = plugin;
        this.watchedFolder = folder;
        this.startWatching();
    }
    
    startWatching() {
        // íŒŒì¼ ìƒì„± ê°ì§€
        this.plugin.registerEvent(
            this.plugin.app.vault.on('create', (file) => {
                if (file instanceof TFile && 
                    file.path.startsWith(this.watchedFolder)) {
                    this.onFileCreated(file);
                }
            })
        );
        
        // íŒŒì¼ ìˆ˜ì • ê°ì§€
        this.plugin.registerEvent(
            this.plugin.app.vault.on('modify', (file) => {
                if (file instanceof TFile && 
                    file.path.startsWith(this.watchedFolder)) {
                    this.onFileModified(file);
                }
            })
        );
    }
    
    async onFileCreated(file: TFile) {
        // ìƒˆ íŒŒì¼ì— í…œí”Œë¦¿ ì ìš©
        const template = `---
created: ${new Date().toISOString()}
tags: []
---

# ${file.basename}
`;
        
        const content = await this.plugin.app.vault.read(file);
        if (!content) {
            await this.plugin.app.vault.modify(file, template);
        }
    }
    
    async onFileModified(file: TFile) {
        // ìˆ˜ì • ì‹œê° ì—…ë°ì´íŠ¸
        const content = await this.plugin.app.vault.read(file);
        const updated = content.replace(
            /modified: .*/,
            `modified: ${new Date().toISOString()}`
        );
        
        if (updated !== content) {
            await this.plugin.app.vault.modify(file, updated);
        }
    }
}

class Logger {
    private prefix: string;
    private debug: boolean;
    
    constructor(prefix: string, debug = false) {
        this.prefix = prefix;
        this.debug = debug;
    }
    
    log(message: string, ...args: any[]) {
        if (this.debug) {
            console.log(`[${this.prefix}]`, message, ...args);
        }
    }
    
    warn(message: string, ...args: any[]) {
        console.warn(`[${this.prefix}]`, message, ...args);
    }
    
    error(message: string, error?: Error) {
        console.error(`[${this.prefix}]`, message, error);
        if (error?.stack) {
            console.error(error.stack);
        }
    }
    
    time(label: string) {
        if (this.debug) {
            console.time(`[${this.prefix}] ${label}`);
        }
    }
    
    timeEnd(label: string) {
        if (this.debug) {
            console.timeEnd(`[${this.prefix}] ${label}`);
        }
    }

interface WordCountSettings {
    showInStatusBar: boolean;
    wordsPerMinute: number;
    showReadingTime: boolean;
}

export default class WordCountPlugin extends Plugin {
    settings: WordCountSettings;
    statusBarItem: HTMLElement;
    
    async onload() {
        // ì„¤ì • ë¡œë“œ
        await this.loadSettings();
        
        // ìƒíƒœë°” ì•„ì´í…œ ì¶”ê°€
        if (this.settings.showInStatusBar) {
            this.statusBarItem = this.addStatusBarItem();
        }
        
        // ì—ë””í„° ë³€ê²½ ì´ë²¤íŠ¸ ë“±ë¡
        this.registerEvent(
            this.app.workspace.on('editor-change', () => {
                this.updateWordCount();
            })
        );
        
        // íŒŒì¼ ì—´ê¸° ì´ë²¤íŠ¸
        this.registerEvent(
            this.app.workspace.on('file-open', () => {
                this.updateWordCount();
            })
        );
        
        // ëª…ë ¹ì–´ ì¶”ê°€
        this.addCommand({
            id: 'show-word-count',
            name: 'Show word count',
            callback: () => {
                this.showWordCountNotice();
            }
        });
    }
    
    updateWordCount() {
        const view = this.app.workspace.getActiveViewOfType(MarkdownView);
        if (!view) return;
        
        const editor = view.editor;
        const text = editor.getValue();
        
        const wordCount = this.countWords(text);
        const charCount = text.length;
        const readingTime = Math.ceil(wordCount / this.settings.wordsPerMinute);
        
        if (this.statusBarItem) {
            let statusText = `Words: ${wordCount}`;
            if (this.settings.showReadingTime) {
                statusText += ` | ${readingTime} min read`;
            }
            this.statusBarItem.setText(statusText);
        }
    }
    
    countWords(text: string): number {
        // ë§ˆí¬ë‹¤ìš´ ë¬¸ë²• ì œê±°
        text = text.replace(/```[\s\S]*?```/g, ''); // ì½”ë“œ ë¸”ë¡

        // ë‹¨ì–´ ìˆ˜ ê³„ì‚°
        const words = text.match(/\b\w+\b/g);
        return words ? words.length : 0;
    }
    
    showWordCountNotice() {
        const view = this.app.workspace.getActiveViewOfType(MarkdownView);
        if (!view) {
            new Notice('No active document');
            return;
        }
        
        const text = view.editor.getValue();
        const wordCount = this.countWords(text);
        const readingTime = Math.ceil(wordCount / this.settings.wordsPerMinute);
        
        new Notice(`
Words: ${wordCount}
Characters: ${text.length}
Reading time: ${readingTime} minutes
        `);
    }
}
